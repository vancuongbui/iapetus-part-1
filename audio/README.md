1. Icons:
    - install:
        npm install react-native-elements --save
        npm react-native link
        npm install react-native-vector-icons --save
        npm react-native link 
    

2. Audio:
    npm install react-native-video  save
    npm install -g rnpm
    rnpm link react-native-video
